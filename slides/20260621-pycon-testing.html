                                                                     Testing Principles                                                                     
                                                                                                                                                                              
                                                                                                                                                                              
                                                                                                                                                                              
              ██ Basic Principles:                                                                                                                                            
                                                                                                                                                                              
                                                                                                                                                                              
                                                           # Arrange: set up your environment, variables, data                                                                
                                                           x, y = 2, 5                                                                                                        
                                                                                                                                                                              
                                                           # Act: execute the function, behaviour you are testing                                                             
                                                           z = add(x, y)                                                                                                      
                                                                                                                                                                              
                                                           # Assert: check the outcomes of your action                                                                        
                                                           assert z == 7
